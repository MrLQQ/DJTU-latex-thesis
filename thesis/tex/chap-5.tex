完全手动完成的表格，如表\ref{tab:tab1}所示。
\begin{table}[htbp] % htbp 浮动优先级
	\centering  % 居中
	\caption{一个表格}  % 表格标题
	\label{tab:tab1}  % 用于引用的label
	% 字母的个数对应列数，|代表分割线
	% l代表左对齐，c代表居中，r代表右对齐
	\begin{tabular}{|c|c|c|c|}   
		\hline  % 表格的横线 
		1 & 2 & 3 & 4 \\  % 表格中的内容，用&分开，\\表示下一行
		\hline 
		0.1 & 0.2 & 0.3 & 0.4 \\
		\hline
	\end{tabular}
\end{table}

嵌套表格，如表\ref{tab:tab3}所示。

    \label{tab:tab3}

        a & b & c \\ \hline

            e & f \\
        \end{tabular}}

    \end{tabular}
